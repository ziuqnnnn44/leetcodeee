class Solution {
    public List<Double> averageOfLevels(TreeNode root) {
        List<Double> res = new ArrayList<>();
        Queue<TreeNode> q = new LinkedList<>();
        
        q.offer(root);
        while(!q.isEmpty()){
            int n = q.size();
            double sum = 0.0;
            for(int i = 0; i < n; i++){
                TreeNode no = q.poll();
                sum += no.val;
                if(no.left != null) q.offer(no.left);
                if(no.right != null) q.offer(no.right);
            }
            res.add(sum/n);
            
        }
        return res;
    }
}
